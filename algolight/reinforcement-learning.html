N = 10000
d = 10
ads_selected = []
number_of_selections = [0]*d
sum_of_rewards = [0]*d
total_reward = 0
for n in range(0,N):
    ad = 0
    max_upper_bound = 0
    for i in range(0,d):
        if number_of_selections[i] > 0:
            average_reward = sum_of_rewards[i] / number_of_selections[i]
            delta_i = math.sqrt((3/2)*(math.log(n+1)/number_of_selections[i]))
            upper_bound = average_reward + delta_i
        else:
            upper_bound = 1e400
